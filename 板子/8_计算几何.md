### 8.1 计算几何基础知识

```C++
#include <iostream>
#include <cmath>
#define endl "\n"
using namespace std;

const double eps = 1e-8;
const int N = 32;

/*
返回值：
	负数 --> -1
	零   --> 0
	正数 --> 1
*/
int sign(double x)
{
	if(fabs(x) < eps) return 0;
	if(x < 0) return -1;
	return 1;
}

/*
返回值：
	x == y --> 0
	x < y  --> -1
	x > y  --> 1
*/
int cmp(double x, double y)
{
	if(fabs(x - y) < eps) return 0;
	if(x < y) return -1;
	return 1;
}

//二维坐标类
struct Point
{
	int x, y;
	Point() {}
	Point(double xx, double yy) : x(xx), y(yy) {}
	Point operator + (const Point &b)
	{
		return Point(x + b.x, y + b.y);
	}
	Point operator - (const Point &b)
	{
		return Point(x - b.x, y - b.y);
	}
	Point operator * (const double &t)
	{
		return Point(t * x, t * y);
	}
	bool operator == (const Point &b)
	{
		return !cmp(x, b.x) && !cmp(y, b.y);
	}
};

//点积 A·B = |A||B|cos(C)
double dot(Point a, Point b)
{
	return a.x * b.x + a.y * b.y;	
}

//叉积 AxB = |A||B|sin(C)
double cross(Point a, Point b)
{
	return a.x * b.y - b.x * a.y;	
}

//计算向量的模
double get_length(Point a)
{
	return sqrt(dot(a, a));	
}

//计算向量夹角
double get_angle(Point a, Point b)
{
	return acos(dot(a, b) / get_length(a) / get_length(b));	
}

//计算两个向量构成的平行四边形的有向面积
double area(Point a, Point b, Point c)
{
	return cross(b - a, c - a);	
}

Point rotate(Point a, double angle)
{
	return Point(a.x * cos(angle) + a.y * sin(angle), 
			-a.x * sin(angle) + a.y * cos(angle));	
}

//判断点在直线上 A x B = 0

//cross(v, w) == 0 两直线平行或重合
//否则
typedef Point Vector;
Point get_line_intersection(Point p, Vector v, Point q, Vector w)
{
	Vector u = p - q;
	double t = cross(w, u) / cross(v, w);
	return p + v * t;
}

//点到直线的距离
double distance_to_line(Point p, Point a, Point b)
{
	Vector v1 = b - a, v2 = p - a;
	return fabs(cross(v1, v2) / get_length(v1));
}

//点到线段的距离
double distance_to_segement(Point p, Point a, Point b)
{
	if(a == b) return get_length(p - a);
	Vector v1 = b - a, v2 = p - a, v3 = p - b;
	if(sign(dot(v1, v2)) < 0) return get_length(v2);
	if(sign(dot(v1, v3)) > 0) return get_length(v3);
	return distance_to_line(p, a, b);	
}

//点在直线上的投影
Point get_line_projection(Point p, Point a, Point b)
{
	Vector v = b - a;
	return a + v * (dot(v, p - a) / dot(v, v));	
}

//点是否在线段上
bool on_segment(Point p, Point a, Point b)
{
	return sign(cross(p - a, p - b)) == 0 
			&& sign(dot(p - a, p - b)) <= 0;
}

//?判断两条线段是否相交
bool segment_intersection(Point a1, Point a2, Point b1, Point b2)
{
	double c1 = cross(a2 - a1, b1 - a1); 
	double c2 = cross(a2 - a1, b2 - a1);
	double c3 = cross(b2 - b1, a1 - b1);
	double c4 = cross(b2 - b1, a2 - b1);
	return sign(c1) * sign(c2) <= 0 &&
			sign(c3) * sign(c4) <= 0;
}

//计算三个点确定的圆的圆心并返回，不存在返回{1e18, 1e18}
typedef pair<double, double> PDD;
auto f = [&](PDD x, PDD y, PDD z) {
	double dy1 = x.y - y.y, dx1 = x.x - y.x;
	double dy2 = x.y - z.y, dx2 = x.x - z.x;
	if(fabs(dy1 * dx2 - dx1 * dy2) < eps) {
		return PDD(1e18, 1e18);
	}
	double a = x.x - y.x;
	double b = x.y - y.y;
	double c = x.x - z.x;
	double d = x.y - z.y;
	double e = ((x.x * x.x - y.x * y.x) - (y.y * y.y - x.y * x.y)) / 2.0;
	double f = ((x.x * x.x - z.x * z.x) - (z.y * z.y - x.y * x.y)) / 2.0;
	double X = (e * d - b * f) / (a * d - b * c);
	double Y = (a * f - e * c) / (a * d - b * c);
	return PDD(X, Y);
};
```

### 8.2 凸包
```C++
typedef pair<double, double> PDD;
PDD operator - (PDD a, PDD b) {
	return PDD(a.x - b.x, a.y - b.y);	
}
struct tubao {
	int n, top = 0;
	vector<PDD> p;
	vector<int> stk;
	tubao() {}
	tubao(int sz, const vector<PDD> &v) : n(sz) {
		p.resize(n);
		stk.resize(2 * n + 5);
		for(int i = 0; i < n; i++) {
			p[i] = {v[i].x, v[i].y};
		}
	}
	//计算向量 a、b 的叉积
	double cross(PDD a, PDD b) {
		return a.x * b.y - b.x * a.y;	
	}
	//计算 a --> b 和 a --> c 的向量构成的四边形的有向面积
	//用于判断 a --> c 在 a --> b 的逆时针方向（面积大于 0）或者顺时针方向（面积小于 0）
	//和 sin函数 在第一、四象限的符号一致
	double area(PDD a, PDD b, PDD c) {
		return cross(b - a, c - a);	
	}		
	//计算两个点的距离
	double dis(int i, int j) {
		double dx = p[i].x - p[j].x;
		double dy = p[i].y - p[j].y;
		return sqrt(dx * dx + dy * dy);	
	}
	//凸包算法：求出凸包上的点[1, top]
	//注意 0 入栈了两次，分别位于下标 1 和 top
	//注意 while 循环里的 < 取等号的时候如果有多个点共线时只会保留线段的端点
	//根据题目要求是否取等号
	void andrew() {
		sort(p.begin(), p.end());
		for(int i = 0; i < n; i++) {
			//先求下凸包
			while(top > 1 && area(p[stk[top - 1]], 
				p[stk[top]], p[i]) < 0) top--;
			stk[++top] = i;
		}	
		//再求上凸包
		int j = top;
		//注意这里从 n - 2 开始，并且 0 会再次入栈
		for(int i = n - 2; i >= 0; i--) {
			while(top > j && area(p[stk[top - 1]],
				p[stk[top]], p[i]) < 0) top--;
			stk[++top] = i;
		}	
	}
	//返回凸包周长
	double get_len() {
		double res = 0;
		for(int i = 2; i <= top; i++) res += dis(stk[i], stk[i - 1]);
		return res;
	}
};
```
