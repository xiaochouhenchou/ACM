### 1.1 启发式合并
#### 1.1.1 普通启发式
##### [洛谷P3201 [HNOI2009] 梦幻布丁](https://www.luogu.com.cn/problem/P3201)

询问一：将某种颜色变成另一种颜色
询问二：求不同颜色段的数量
```C++
#include <iostream>
#include <cstring>
#define endl "\n"
using namespace std;

const int N = 100010, M = N * 10;

int n, m, ans;
int h[M], e[N], ne[N], idx;
int c[N], cnt[M], p[M], sz[M];

void add(int a, int b)
{
	e[idx] = b, ne[idx] = h[a], h[a] = idx++;
	sz[a]++;
}

void merge(int &x, int &y)
{
	if(x == y) return;
	if(sz[x] > sz[y]) swap(x, y);
	
	for(int i = h[x]; i != -1; i = ne[i])
	{
		int j = e[i];
		ans -= c[j - 1] == y;
		ans -= c[j + 1] == y;
	}
	
	for(int i = h[x]; i != -1; i = ne[i])
	{
		int j = e[i];
		c[j] = y;
		
		if(ne[i] == -1)
		{
			ne[i] = h[y];
			h[y] = h[x];
			h[x] = -1;
			break;
		}
	}
	
	sz[y] += sz[x], sz[x] = 0;
}

int main()
{
	ios::sync_with_stdio(false), cin.tie(0), cout.tie(0);
	memset(h, -1, sizeof h);
	for(int i = 0; i < M; i++) p[i] = i;
	cin >> n >> m;
	for(int i = 1; i <= n; i++) 
	{
		cin >> c[i];
		add(c[i], i);
		ans += c[i] != c[i - 1];
	}
	while(m--)
	{
		int op;
		cin >> op;
		if(op - 1) cout << ans << endl;
		else 
		{
			int x, y;
			cin >> x >> y;
			merge(p[x], p[y]);
		}
	}
	return 0;
}
```

#### 1.1.2 树上启发式
##### [CF600E. Lomsat gelral](https://www.luogu.com.cn/problem/CF600E)

**题意**：求子树中颜色出现次数最多的颜色之和。

```C++
#include <iostream>
#define endl "\n"
using namespace std;

typedef long long LL;
const int N = 100010, M = N * 2;

int n;
int h[N], e[M], ne[M], idx;
int c[N], cnt[N], sz[N], son[N], mx;
LL sum, ans[N];

void add(int a, int b)
{
	e[idx] = b, ne[idx] = h[a], h[a] = idx++;
}

//求重儿子
void dfs1(int u, int fa)
{
	sz[u] = 1;	
	for(int i = h[u]; i != -1; i = ne[i])
	{
		int j = e[i];
		if(j == fa) continue;
		dfs1(j, u);
		sz[u] += sz[j];
		if(sz[son[u]] < sz[j]) son[u] = j;
	}
}

//更新cnt
void update(int u, int fa, int v, int son)
{
	int t = c[u];
	cnt[t] += v;
	if(cnt[t] > mx) mx = cnt[t], sum = t;
	else if(cnt[t] == mx) sum += t;
	
	for(int i = h[u]; i != -1; i = ne[i])
	{
		int j = e[i];
		if(j == fa || j == son) continue;
		update(j, u, v, son);
	}
}

//统计cnt
void dfs2(int u, int fa, int op)
{
	for(int i = h[u]; i != -1; i = ne[i])
	{
		int j = e[i];
		if(j == fa || j == son[u]) continue;
		dfs2(j, u, 0);
	}	
	
	if(son[u]) dfs2(son[u], u, 1);
	update(u, fa, 1, son[u]);
	
	ans[u] = sum;
	if(!op) update(u, fa, -1, 0), sum = mx = 0;
}

int main()
{
	ios::sync_with_stdio(false), cin.tie(0), cout.tie(0);
	cin >> n;
	for(int i = 1; i <= n; i++) cin >> c[i], h[i] = -1;
	for(int i = 1; i <= n - 1; i++)
	{
		int a, b;
		cin >> a >> b;
		add(a, b), add(b, a);
	}
	
	dfs1(1, -1);
	dfs2(1, -1, 1);
	
	for(int i = 1; i <= n; i++) 
		cout << ans[i] << (i != n ? " " : endl);
	return 0;
}
```

```C++
void quick_sort(int q[], int l, int r) {
    if(l >= r) {
        return;
    }
    int x = q[(l + r) >> 1], i = l - 1, j = r + 1;
    while(i < j) {
        do i++; while(q[i] < x);
        do j--; while(q[j] > x);
        if(i < j) {
            swap(q[i], q[j]);
        }
    }
    quick_sort(q, l, j);
    quick_sort(q, j + 1, r);
}
```

**快速排序求第 $k$ 大的数**
```C++
#include <bits/stdc++.h>
using namespace std;

const int N = 100010;

int n, k;
int q[N];

int quick_sort(int l, int r, int k) {
    if(l >= r) {
        return q[l];
    }
    int x = q[(l + r) / 2], i = l - 1, j = r + 1;
    while(i < j) {
        while(q[++i] < x);
        while(q[--j] > x);
        if(i < j) {
            swap(q[i], q[j]);
        }
    }
    int sl = j - l + 1;
    if(sl < k) {
        return quick_sort(j + 1, r, k - sl);
    }
    else {
        return quick_sort(l, j, k);
    }
}
```

**归并排序求逆序对**

```C++
#include <bits/stdc++.h>
using namespace std;

typedef long long LL;
const int N = 100010;

int n;
int a[N], tmp[N];


LL merge_sort(int q[], int l, int r) {
    if(l >= r) {
        return 0;
    }
    int mid = l + r >> 1, k = 0;
    LL res = merge_sort(q, l, mid) + merge_sort(q, mid + 1, r);
    int i = l, j = mid + 1;
    while(i <= mid && j <= r) {
        if(q[i] <= q[j]) {
            tmp[k++] = q[i++];
        }
        else {
            res += mid - i + 1;
            tmp[k++] = q[j++];
        }
    }
    while(i <= mid) tmp[k++] = q[i++];
    while(j <= r) tmp[k++] = q[j++];
    for(int i = l, j = 0; i <= r; i++, j++) {
        q[i] = tmp[j];
    }
    return res;
}


int main() {
    cin >> n;
    for(int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }
    cout << merge_sort(a, 0, n - 1) << endl;
    return 0;
}
```
