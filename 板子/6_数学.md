### 6.1 快速幂
#### [模板]()
```C++
#include <iostream>
#include <cstdio>
using namespace std;


typedef long long LL;

LL qmi(int a, int b, int p) {
    LL res = 1;
    while(b) {
        if(b & 1) res = res * a % p;
        b >>= 1;
        a = (LL)a * a % p;
    }
    return res;
}

int main() {
    int n;
    scanf("%d", &n);
    while(n--) {
        int a, b, p;
        scanf("%d%d%d", &a, &b, &p);
        printf("%lld\n", qmi(a, b, p));
    }
    return 0;
}
```

### 6.2 素数
#### [筛质数模板]()
```C++
#include <iostream>
#include <cstdio>
using namespace std;

const int N = 1e6 + 10;

int n, cnt;
int prime[N];
bool st[N];

void pri(int n) {
    for(int i = 2; i <= n; i++) {
        if(!st[i]) prime[cnt++] = i;
        for(int j = 0; prime[j] <= n / i; j++) {
            st[prime[j] * i] = true;
            if(i % prime[j] == 0) break;  //prime从小到大枚举，则 prime[j] < prime[j + 1];
                                                    //如果 prime[j] 可以被i整除，那么很显然（prime[j + 1]*i）的最小质因数是 prime[j] 而不是 prime[j + 1]
                                                   // 再继续用 prime[j + 1] 筛，就不满足用一个数的最小质因数筛掉它这个条件了。
        }
    }
    return;
} 

int main() {
    scanf("%d", &n);
    pri(n);
    printf("%d\n", cnt);
    return 0;
}
```
#### [洛谷P3601 签到题——筛法、欧拉函数](https://www.luogu.com.cn/problem/P3601)

我们可以知道 $\sum_{i = l}^{r}qiandao(i) \space mod\space 666623333 = \sum_{i = l}^r i - \sum_{i = l}^{r} \phi (i)$，其中 $\phi(i)$ 为欧拉函数，表示 $[1, i]$ 中和 $i$ 互质的数的数量。

如果对 $[l, r]$ 中的数直接分解质因数求欧拉函数的话，因为数据范围较大，会 $TLE$。因此我们可以先把所有小于等于 $10^6$ 内的质数筛出来（$\sqrt{r} \le 10^6$），然后直接把这些质数的倍数筛出来，并计算它们的欧拉函数（类似分解质因数的做法，这里的质数的倍数的质因数就是当前的质数）。当然要注意当某个数本身就是一个比较大的质数时，要在最后再处理一下。

注意数据范围较大，开不了这么大的数据，因此 $i$ 这个数的下标可以从 $i - l + 1$ 开始。

**代码**
```C++
#include <iostream>
#define endl "\n"
using namespace std;

const int N = 1000010, mod = 666623333;
typedef long long LL;

LL l, r, cnt;
LL p[N], phi[N], ls[N];
bool st[N];

//线性筛
void init(int n)
{
	for(int i = 2; i <= n; i++)
	{
		if(!st[i]) p[cnt++] = i;
		for(int j = 0; p[j] <= n / i; j++)
		{
			st[p[j] * i] = true;
			if(i % p[j] == 0) break;
		}
	}
}

int main()
{
	ios::sync_with_stdio(false), cin.tie(0), cout.tie(0);
	init(N - 1);
	cin >> l >> r;
	//i + l - 1这个数映射到 i 这个下标
	for(LL i = 1; i <= r - l + 1; i++) phi[i] = ls[i] = i + l - 1;
	//枚举质数去筛掉它们的倍数
	for(int i = 0; i < cnt; i++)
	{
		//上取整
		LL t = (l + p[i] - 1) / p[i];
		//t = 1的时候需要加 1，不然会把质数自己筛掉
		if(t == 1) t++;
		
		for(LL j = t * p[i]; j <= r; j += p[i])
		{
			//j映射到数组的下标
			int cur = j - l + 1;
			//计算欧拉函数
			phi[cur] = phi[cur] / p[i] * (p[i] - 1);
			//将p[i] 这个质因子除干净
			while(ls[cur] % p[i] == 0) ls[cur] /= p[i];
		}
	}
	LL ans = 0;
	for(LL i = l; i <= r; i++)
	{
		int cur = i - l + 1;
		//i 还剩下一个比较大的质数没除干净
		if(ls[cur] > 1) phi[cur] = phi[cur] / ls[cur] * (ls[cur] - 1);
		ans = (ans + i - phi[cur]) % mod;
	}
	cout << ans << endl;
	return 0;
}
```

### 6.3 约数
#### [约数个数]()
$ans = (k_1 + 1) * (k_2 + 1) * ... * (k_n + 1)$
```C++
#include <iostream>
#include <cstdio>
#include <unordered_map>
using namespace std;

const int mod = 1e9 + 7;

int n;
typedef long long LL;

int main() {
    scanf("%d", &n);
    LL res = 1;
    unordered_map<int, int> primes;
    while(n--) {
        int x;
        scanf("%d", &x);
        for(int i = 2; i <= x / i; i++) {
            if(x % i == 0) {
                while(x % i == 0) {
                    x /= i;
                    primes[i]++;
                }
            }
        }
        if(x > 1)primes[x]++;
    }
    for(auto x : primes) {
        int p = x.second;
        res = res * (p + 1) % mod;
    }
    printf("%lld", res);
    return 0;
 }
```

#### [约数之和]()
$ans = (p_{1} ^ {0} + p_{1} ^ {1} + ... + p_{1} ^ {k_1}) * ... * (p_{n} ^ {0} + p_{n} ^ {1} + ... + p_{n} ^ {k_n})$
```C++
#include <iostream>
#include <cstdio>
#include <unordered_map>
using namespace std;

const int N = 1e9 + 7;
typedef long long LL;

int n;

int main() {
    scanf("%d", &n);
    LL res = 1;
    unordered_map<int, int> primes;
    while(n--) {
        int x;
        scanf("%d", &x);
        for(int i = 2; i <= x / i; i++) {
            if(x % i == 0) {
                while(x % i == 0) {
                    x /= i;
                    primes[i]++;
                }
            }
        }
        if(x > 1)primes[x]++;
    }
    for(auto x : primes) {
        int p = x.first, a = x.second;
        LL t = 1;
        while(a--) {
            t = (t * p + 1) % N;
        }
        res = res * t % N;
    }
    printf("%lld", res);
    return 0;
}
```

### 6.4 欧拉函数
#### [模板]()
```C++
#include <iostream>
#include <cstdio>
using namespace std;

typedef long long LL;

LL phi(int n) {
    LL res = n;
    for(int i = 2; i <= n / i; i++) {
        if(n % i == 0) {
            res = res / i * (i - 1);
            while(n % i == 0) n /= i;
        }
    }
    if(n > 1) res = res / n * (n - 1);
    return res;
}

int main() {
    int n;
    scanf("%d", &n);
    while(n--) {
        int a;
        scanf("%d", &a);
        printf("%lld\n", phi(a));
    }
    return 0;
}
```

#### [1~n中的数的欧拉函数之和]()
```C++
#include <iostream>
#include <cstdio>

const int N = 1e6 + 10;
typedef long long LL;

int primes[N], cnt;
int phi[N];
bool st[N];

LL get_phi(int n) {
    phi[1] = 1;
    for(int i = 2; i <= n; i++) {
        if(!st[i]) {
            phi[i] = i - 1;
            primes[cnt++] = i;
        }
        for(int j = 0; primes[j] <= n / i; j++) {
            st[primes[j] * i] = true;
            if(i % primes[j] == 0) {
                phi[primes[j] * i] = phi[i] * primes[j];
                break;
            }
            phi[primes[j] * i] = phi[i] * (primes[j] - 1);
        }
    }
    LL res = 0;
    for(int i = 1; i <= n; i++) {
        res += phi[i];
    }
    return res;
}

int main() {
    int n;
    scanf("%d", &n);
    printf("%lld\n", get_phi(n));
    return 0;
}
```

### 6.5 组合数
#### [小范围求组合数]()
```C++
#include <iostream>
#include <cstdio>
using namespace std;

typedef long long LL;
const int N = 2010, mod = 1e9 + 7;

int n;
int c[N][N];

int main() {
    scanf("%d", &n);
    for(int i = 0; i < N; i++) {
        for(int j = 0; j <= i; j++) {
            if(j == 0) c[i][j] = 1;
            else c[i][j] = (LL)(c[i - 1][j] + c[i - 1][j - 1]) % mod;
        }
    }
    while(n--) {
        int a, b;
        scanf("%d%d", &a, &b);
        printf("%d\n", c[a][b]);
    }
    return 0;
}
```

#### [较大范围求组合数]()
```C++
//求组合数类
//由于用到了逆元，因此只能用于 mod 为质数的情况
struct Com {
	LL n, mod;
	vector<LL> fac;
	vector<LL> infac;
	Com() {}
	Com(const LL &sz, const LL &M) : n(sz), mod(M) {
		fac.resize(n + 1);
		infac.resize(n + 1);
		fac[0] = infac[0] = 1;
		for(LL i = 1; i <= n; i++) {
			fac[i] = fac[i - 1] * i % mod;
			infac[i] = infac[i - 1] * qmi(i, mod - 2) % mod;
		}
	}
	LL qmi(LL a, LL b) {
		LL res = 1;
		while(b) {
			if(b & 1) {
				res = res * a % mod;
			}
			b >>= 1;
			a = a * a % mod;
		}
		return res % mod;
	}
	LL C(LL a, LL b) {
		if(a < 0 || b < 0 || a < b) {
			return 0;
		}
		return fac[a] * infac[a - b] % mod * infac[b] % mod;
	}
};
```

#### [SG函数]()
```C++
#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <unordered_set>
using namespace std;

const int N = 110, M = 10010;

int n, k;
int s[N], f[M];

int sg(int x) {
    if(f[x] != -1) return f[x];
    
    unordered_set<int> p;
    for(int i=  0; i < k; i++) {
        if(x >= s[i]) p.insert(sg(x - s[i]));
    }
    for(int i = 0; ; i++) {
        if(!p.count(i)) return f[x] = i;
    }
    return 0;
}

int main() {
    scanf("%d", &k);
    for(int i = 0; i < k; i++) {
        scanf("%d", &s[i]);
    }
    scanf("%d", &n);
    memset(f, -1, sizeof f);
    int res = 0;
    while(n--) {
        int x;
        scanf("%d", &x);
        res ^= sg(x);
    }
    if(res) puts("Yes");
    else puts("No");
    return 0;
}
```

### 6.7 龟速乘
#### [模板]()
```C++
#include <iostream>
#define endl "\n"
using namespace std;

typedef long long LL;

LL a, b, p;

LL qm()
{
    LL res = 0;
    while(b)
    {
        if(b & 1) res = (res + a) % p;
        a = a * 2 % p;
        b >>= 1;
    }
    return res;
}

int main()
{
    cin >> a >> b >> p;
    cout << qm() << endl;
    return 0;
}
```
